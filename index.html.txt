<script type="text/babel">
    const { useState } = React;

    // --- Helper Components ---
    const InputNumber = ({ label, value, onChange, unit, badge, color = "text-gray-700", placeholder }) => (
        <div className="relative">
            <label className="block text-xs font-bold text-gray-500 mb-1">{label}</label>
            <div className="relative flex items-center">
                <input 
                    type="number" 
                    value={value || ''} 
                    onChange={onChange} 
                    placeholder={placeholder}
                    className={w-full border border-gray-300 p-2 rounded-lg focus:ring-2 focus:ring-blue-500 outline-none text-left font-bold dir-ltr pl-3 ${badge ? 'pr-14' : 'pr-3'} ${color}}
                />
                {badge && <span className="absolute right-1 bg-blue-100 text-blue-800 text-[9px] px-1.5 py-0.5 rounded font-bold">{badge}</span>}
                {unit && !badge && <span className="absolute right-2 text-[10px] text-gray-400">{unit}</span>}
            </div>
        </div>
    );

    const M2 = () => <span className="font-eng text-[9px] inline-block dir-ltr">m<sup>2</sup></span>;

    const App = () => {
        const [step, setStep] = useState(1); 

        // --- Data State ---
        const [land, setLand] = useState({
            code: 'A-1403',
            area: 300,
            floors: 6, 
            occupancy: 60,
            commonAreaPerFloor: 20, 
            // Prices inputs (x1000 logic handled in UI, stored as raw input)
            pricePerMeterInput: 100000, 
            totalLandPriceInput: 30000000, // Calculated init: 300 * 100000
            sellPriceInput: 65000,     
            buildCostPerMeterInput: 27000 
        });

        const [frontages, setFrontages] = useState([
            { id: 1, name: 'بر اصلی', len: 12, cons: 1.2 },
            { id: 2, name: 'بر دوم', len: 0, cons: 0 },
            { id: 3, name: 'بر سوم', len: 0, cons: 0 },
            { id: 4, name: 'بر چهارم', len: 0, cons: 0 },
        ]);

        const [materials, setMaterials] = useState([
            { id: 1, name: 'میلگرد', unit: 'کیلوگرم', coeff: 50, price: 40000 },
            { id: 2, name: 'میلگرد حرارتی', unit: 'کیلوگرم', coeff: 3, price: 40000 },
            { id: 3, name: 'بتن', unit: 'متر مکعب', coeff: 0.4, price: 4600000 },
            { id: 4, name: 'تیرچه', unit: 'متر طول', coeff: 1.3, price: 200000 },
            { id: 5, name: 'یونولیت', unit: 'عدد', coeff: 0.68, price: 500000 },
        ]);

        // --- Logic Handlers ---
        
        // Handle Land Price Logic (Total <-> Per Meter)
        const updateLandPrice = (type, val) => {
            const numVal = Number(val);
            if (type === 'meter') {
                // User entered Price Per Meter -> Calculate Total
                // Input is x1000, so total is x1000 too for consistency in inputs
                setLand(prev => ({
                    ...prev,
                    pricePerMeterInput: numVal,
                    totalLandPriceInput: numVal * prev.area
                }));
            } else {
                // User entered Total Price -> Calculate Per Meter
                setLand(prev => ({
                    ...prev,
                    totalLandPriceInput: numVal,
                    pricePerMeterInput: prev.area > 0 ? numVal / prev.area : 0
                }));
            }
        };

        const updateArea = (val) => {
            const numVal = Number(val);
            setLand(prev => ({
                ...prev,
                area: numVal,
                // Recalculate Total Price based on existing rate
                totalLandPriceInput: numVal * prev.pricePerMeterInput
            }));
        };

        // --- Calculation Engine ---
        const calculate = () => {
            const realPricePerMeter = land.pricePerMeterInput * 1000;
            const realSellPrice = land.sellPriceInput * 1000;
            const realBuildCostPerMeter = land.buildCostPerMeterInput * 1000;

            // 1. Console Area (One Floor)
            let consoleAreaOneFloor = 0;
            frontages.forEach(f => consoleAreaOneFloor += (f.len * f.cons));

            // 2. Base Area
            const baseAreaOneFloor = land.area * (land.occupancy / 100);

            // 3. Gross (Non-Useful)
            const totalBaseArea = baseAreaOneFloor * (land.floors + 1);
            const totalConsoleArea = consoleAreaOneFloor * land.floors;
            const totalGross = totalBaseArea + totalConsoleArea; // Total Non-Useful

            // 4. Useful
            const usefulOneFloor = (baseAreaOneFloor - land.commonAreaPerFloor) + consoleAreaOneFloor;
            const totalUseful = usefulOneFloor * land.floors;
            const totalCommon = land.commonAreaPerFloor * land.floors; // NEW ITEM

            // 5. Financials
            const totalBuildCost = totalGross * realBuildCostPerMeter; // NEW ITEM: Cost * Gross
            const totalLandCost = land.area * realPricePerMeter; // NEW ITEM
            
            // Total Project Cost (Finished)
            const totalProjectCost = totalLandCost + totalBuildCost; 
            
            // Finished Cost Per Useful Meter (Land + Build)
            const finishedPricePerMeter = totalUseful > 0 ? totalProjectCost / totalUseful : 0; // NEW ITEM

            const totalRevenue = totalUseful * realSellPrice; 
            const netProfit = totalRevenue - totalProjectCost;
            const profitMargin = totalProjectCost > 0 ? (netProfit / totalProjectCost) * 100 : 0;
            
            const landCostPerUseful = totalUseful > 0 ? totalLandCost / totalUseful : 0;

            // Material Cost
            let matCostTotal = 0;
            const matList = materials.map(m => {
                const qty = totalGross * m.coeff;
                const cost = qty * m.price;
                matCostTotal += cost;
                return { ...m, qty, cost };
            });

            return { 
                totalGross, totalUseful, totalCommon, totalConsoleArea,
                totalBuildCost, totalLandCost, totalProjectCost, 
                finishedPricePerMeter, landCostPerUseful,
                totalRevenue, netProfit, profitMargin,
                realPricePerMeter, realSellPrice, realBuildCostPerMeter,
                matList
            };
        };

        const res = calculate();
        const fmt = (n) => Number(n).toFixed(0).replace(/\B(?=(\d{3})+(?!\d))/g, ",");

        const CardexRow = ({ icon, title, value, unit, colorClass, titleColor = "text-gray-500", bgColor="bg-white" }) => (
            <div className={${bgColor} p-1.5 rounded-lg shadow-sm border border-gray-200 flex items-center justify-between}>
                <div className="flex items-center gap-1.5 overflow-hidden">
                    <div className={w-5 h-5 min-w-[1.25rem] rounded-full flex items-center justify-center ${colorClass} bg-opacity-10}>
                        <i className={fa ${icon} ${colorClass} text-[9px]}></i>
                    </div>
                    <span className={${titleColor} font-bold text-[9px] whitespace-nowrap}>{title}</span>
                </div>
                <div className="text-left leading-tight min-w-[60px]">
                    <div className={font-titr text-xs ${colorClass}}>{value}</div>
                    {unit && <div className="text-[7px] text-gray-400">{unit}</div>}
                </div>
            </div>
        );

        return (
            <div className="min-h-screen pb-10 font-sans">
                
                {/* --- Operator Panel --- */}
                <div className="bg-white shadow-sm p-3 mb-4 sticky top-0 z-50 no-print border-b border-gray-200">
                    <div className="max-w-4xl mx-auto flex flex-col md:flex-row justify-between items-center gap-3">
                        <h1 className="font-titr text-blue-900 text-base">سیستم هوشمند کاردکس</h1>
                        <div className="flex bg-gray-100 rounded-lg p-1">
                            {['اطلاعات ملک', 'ضرایب', 'قیمت‌گذاری', 'مشاهده کاردکس'].map((t, i) => (
                                <button key={i} onClick={() => setStep(i + 1)} 
                                    className={px-3 py-1 rounded-md text-xs font-bold transition-all ${step === i + 1 ? 'bg-blue-600 text-white shadow' : 'text-gray-500 hover:text-gray-800'}}>
                                    {t}
                                </button>
                            ))}
                        </div>
                    </div>
                </div>

                <div className="max-w-md mx-auto px-2">
                    
                    {/* --- STEPS 1: Inputs --- */}
                    {step === 1 && (
                        <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-200 animate-fade-in no-print">
                            <h2 className="font-titr text-base text-gray-800 mb-4 flex items-center gap-2">
                                <span className="bg-blue-100 text-blue-600 w-6 h-6 flex items-center justify-center rounded-full text-xs">1</span>
                                مشخصات زمین و تراکم
                            </h2>
                            <div className="grid grid-cols-2 gap-3 mb-4">
                                <InputNumber label="کد فایل" value={land.code} onChange={e=>setLand({...land, code:e.target.value})} />
                                {/* Update Area Logic */}
                                <InputNumber label="مساحت زمین" value={land.area} onChange={e=>updateArea(e.target.value)} unit="متر" />
                                <InputNumber label="تعداد سقف (مسکونی)" value={land.floors} onChange={e=>setLand({...land, floors:Number(e.target.value)})} unit="طبقه" />
                                <InputNumber label="سطح اشغال (%)" value={land.occupancy} onChange={e=>setLand({...land, occupancy:Number(e.target.value)})} unit="%" />
                                <div className="col-span-2 bg-yellow-50 p-2 rounded-lg border border-yellow-200">
                                    <InputNumber label="متراژ مشاعات (هر طبقه)" value={land.commonAreaPerFloor} onChange={e=>setLand({...land, commonAreaPerFloor:Number(e.target.value)})} unit="متر مربع" />
                                </div>
                            </div>
                            <div className="bg-gray-50 p-3 rounded-lg border border-gray-200">
                                <h3 className="font-titr text-xs text-gray-700 mb-2">کنسول‌ها (طول × عمق پیش‌آمدگی)</h3>
                                {frontages.map(f => (
                                    <div key={f.id} className="flex gap-2 items-center mb-2 last:mb-0">
                                        <span className="w-16 text-xs font-bold text-gray-600">{f.name}</span>
                                        <input type="number" placeholder="طول" value={f.len} onChange={e=>{const n=[...frontages];n.find(x=>x.id===f.id).len=Number(e.target.value);setFrontages(n)}} className="border p-1 rounded w-full text-center text-xs"/>
                                        <input type="number" placeholder="عمق" value={f.cons} onChange={e=>{const n=[...frontages];n.find(x=>x.id===f.id).cons=Number(e.target.value);setFrontages(n)}} className="border p-1 rounded w-full text-center text-xs bg-yellow-50"/>
                                    </div>
                                ))}
                            </div>
                            <button onClick={()=>setStep(2)} className="w-full mt-4 bg-blue-600 text-white py-2.5 rounded-lg font-titr text-sm">مرحله بعد</button>
                        </div>
                    )}

                    {step === 2 && (
                        <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-200 animate-fade-in no-print">
                            <h2 className="font-titr text-base text-gray-800 mb-4">ضرایب مصرف</h2>
                            <div className="space-y-2">
                                {materials.map(m => (
                                    <div key={m.id} className="flex justify-between items-center bg-gray-50 p-2 rounded border">
                                        <span className="text-xs font-bold text-gray-700">{m.name}</span>
                                        <input type="number" value={m.coeff} onChange={e=>{const n=[...materials];n.find(x=>x.id===m.id).coeff=Number(e.target.value);setMaterials(n)}} className="border p-1 rounded w-20 text-center font-bold text-xs"/>
                                    </div>
                                ))}
                            </div>
                            <div className="mt-4 flex gap-2">
                                <button onClick={()=>setStep(1)} className="px-4 py-2 rounded-lg bg-gray-100 text-xs font-bold">بازگشت</button>
                                <button onClick={()=>setStep(3)} className="flex-1 bg-blue-600 text-white py-2 rounded-lg font-titr text-sm">مرحله بعد</button>
                            </div>
                        </div>
                    )}

                    {step === 3 && (
                        <div className="bg-white p-4 rounded-xl shadow-sm border border-gray-200 animate-fade-in no-print">
                            <h2 className="font-titr text-base text-gray-800 mb-4">قیمت‌گذاری</h2>
                            <div className="space-y-3">
                                {/* Land Price Calculation Logic */}
                                <div className="bg-blue-50 p-3 rounded-lg border border-blue-100 space-y-3">
                                    <InputNumber 
                                        label="قیمت هر متر زمین" 
                                        value={land.pricePerMeterInput} 
                                        onChange={e=>updateLandPrice('meter', e.target.value)} 
                                        badge="×1000" 
                                    />
                                    <div className="border-t border-blue-200"></div>
                                    <InputNumber 
                                        label="قیمت کل زمین (محاسبه خودکار)" 
                                        value={land.totalLandPriceInput} 
                                        onChange={e=>updateLandPrice('total', e.target.value)} 
                                        badge="×1000" 
                                        color="text-blue-900"
                                    />
                                </div>
                                
                                <div className="bg-green-50 p-3 rounded-lg border border-green-100">
                                    <InputNumber label="فروش نوساز (متر)" value={land.sellPriceInput} onChange={e=>setLand({...land, sellPriceInput:Number(e.target.value)})} badge="×1000" />
                                </div>
                                <div className="bg-amber-50 p-3 rounded-lg border border-amber-100">
                                    <InputNumber label="هزینه ساخت (متر)" value={land.buildCostPerMeterInput} onChange={e=>setLand({...land, buildCostPerMeterInput:Number(e.target.value)})} badge="×1000" color="text-amber-800" />
                                </div>
                            </div>
                            <button onClick={()=>setStep(4)} className="w-full mt-4 bg-green-600 text-white py-3 rounded-lg font-titr text-sm shadow-lg">مشاهده کاردکس</button>
                        </div>
                    )}

                    {/* --- STEP 4: COMPACT CARDEX --- */}
                    {step === 4 && (
                        <div>
                            {/* Buttons REMOVED as requested */}

                            <div className="print-wrapper pt-4">
                                {/* Compact Cardex Container */}
                                <div className="print-area bg-white mx-auto shadow-2xl overflow-hidden relative flex flex-col" 
                                     style={{width: '100%', padding: '12px', borderRadius: '16px'}}>
                                    
                                    {/* 1. Header (Compact) */}
                                    <div className="text-center mb-3">
                                        <div className="flex items-center justify-center gap-2 mb-1">
                                            <div className="w-10 h-10 bg-blue-900 rounded-full flex items-center justify-center shadow border-2 border-white">
                                                <i className="fa fa-building text-sm text-white"></i>
                                            </div>
                                            <h1 className="text-blue-900 font-titr text-lg">مشاورین املاک آراسته</h1>
                                        </div>
                                        <div className="flex justify-center items-center gap-1 text-[10px] text-gray-600 font-bold">
                                            <span>نبش چهارراه داماهی</span>
                                            <span className="text-amber-500 mx-1">•</span>
                                            <span className="dir-ltr font-titr text-xs">09121613067</span>
                                        </div>
                                    </div>

                                    {/* 2. Info Strip */}
                                    <div className="bg-blue-900 text-white rounded-lg p-2 flex justify-between items-center shadow-md mb-2">
                                        <div className="text-center flex-1 border-l border-blue-700">
                                            <div className="text-[8px] text-blue-200">کد فایل</div>
                                            <div className="font-titr text-xs">{land.code}</div>
                                        </div>
                                        <div className="text-center flex-1 border-l border-blue-700">
                                            <div className="text-[8px] text-blue-200">تعداد سقف</div>
                                            <div className="font-titr text-xs">{land.floors}</div>
                                        </div>
                                        <div className="text-center flex-1">
                                            <div className="text-[8px] text-blue-200">متراژ زمین</div>
                                            <div className="font-titr text-xs flex items-center justify-center gap-1 dir-ltr">
                                                <span>{land.area}</span> <M2 />
                                            </div>
                                        </div>
                                    </div>

                                    {/* 3. Physical Stats */}
                                    <div className="grid grid-cols-2 gap-2 mb-2">
                                        <CardexRow icon="fa-layer-group" title="مساحت غیرمفید" value={fmt(res.totalGross)} unit="متر مربع (کل بنا)" colorClass="text-gold-custom" titleColor="text-gold-custom" bgColor="bg-amber-50" />
                                        <CardexRow icon="fa-ruler-combined" title="مساحت مفید" value={fmt(res.totalUseful)} unit="متر مربع (قابل فروش)" colorClass="text-green-600" titleColor="text-green-700" bgColor="bg-green-50" />
                                        <CardexRow icon="fa-th-large" title="مساحت کل مشاعات" value={fmt(res.totalCommon)} unit="متر مربع" colorClass="text-gray-600" titleColor="text-gray-500" />
                                        <CardexRow icon="fa-columns" title="مساحت کنسول‌ها" value={fmt(res.totalConsoleArea)} unit="متر مربع" colorClass="text-gray-600" titleColor="text-gray-500" />
                                    </div>

                                    {/* 4. Financial - Unit Prices */}
                                    <div className="grid grid-cols-2 gap-2 mb-2">
                                        <CardexRow icon="fa-tag" title={<>قیمت زمین (<M2 />)</>} value={fmt(res.realPricePerMeter)} unit="تومان" colorClass="text-blue-900" titleColor="text-blue-900" />
                                        <CardexRow icon="fa-hard-hat" title={<>مبنای هزینه ساخت (<M2 />)</>} value={fmt(res.realBuildCostPerMeter)} unit="تومان" colorClass="text-gold-custom" titleColor="text-gold-custom" />
                                        <CardexRow icon="fa-chart-line" title={<>ارزش فروش (<M2 />)</>} value={fmt(res.realSellPrice)} unit="تومان" colorClass="text-green-600" titleColor="text-green-700" />
                                        <CardexRow icon="fa-check-circle" title={<>تمام شده (<M2 />) (زمین و ساخت)</>} value={fmt(res.finishedPricePerMeter)} unit="تومان" colorClass="text-gold-custom" titleColor="text-gold-custom" />
                                    </div>

                                    {/* 5. Financial - Totals */}
                                    <div className="bg-white p-2 rounded-lg shadow-sm border border-gray-100 mb-2">
                                        <div className="flex justify-between items-center mb-2 border-b border-gray-100 pb-1">
                                            <div className="text-[9px] font-bold text-gray-700">خلاصه هزینه‌های کل پروژه</div>
                                        </div>
                                        
                                        <div className="space-y-1.5 mb-2">
                                            <div className="flex justify-between items-center bg-blue-50 p-1 rounded">
                                                <span className="text-[8px] font-bold text-blue-900">هزینه کل زمین</span>
                                                <span className="font-titr text-[10px] text-blue-900">{fmt(res.totalLandCost)}</span>
                                            </div>
                                            <div className="flex justify-between items-center bg-amber-50 p-1 rounded">
                                                <span className="text-[8px] font-bold text-amber-700">هزینه کل ساخت</span>
                                                <span className="font-titr text-[10px] text-amber-700">{fmt(res.totalBuildCost)}</span>
                                            </div>
                                            <div className="flex justify-between items-center bg-gray-50 p-1 rounded border border-gray-100">
                                                <span className="text-[8px] font-bold text-gray-600">قیمت تمام شده کل (زمین + ساخت)</span>
                                                <span className="font-titr text-[10px] text-gray-800">{fmt(res.totalProjectCost)}</span>
                                            </div>
                                        </div>
                                    </div>

                                    {/* 6. Total Revenue Frame */}
                                    <div className="bg-gradient-to-r from-blue-600 to-blue-500 rounded-lg p-2.5 shadow-md text-white mb-2 relative overflow-hidden">
                                        <div className="absolute top-0 right-0 w-16 h-16 bg-white opacity-10 rounded-full -mr-6 -mt-6 pointer-events-none"></div>
                                        <div className="flex justify-between items-center relative z-10">
                                            <div className="text-[9px] text-blue-100">فروش کل مساحت مفید (تومان)</div>
                                            <div className="font-titr text-base">{fmt(res.totalRevenue)}</div>
                                        </div>
                                    </div>

                                    {/* 7. Net Profit (3D Styled) */}
                                    <div className="bg-3d-texture rounded-xl p-4 shadow-xl relative overflow-hidden text-white mb-2">
                                        <div className="flex justify-between items-center relative z-10">
                                            <div>
                                                <div className="text-[9px] text-blue-200 mb-0.5 font-light">سود خالص پروژه (تومان)</div>
                                                <div className="font-titr text-2xl text-green-neon tracking-wide drop-shadow-md">{fmt(res.netProfit)}</div>
                                            </div>
                                            <div className="text-left bg-white/10 px-3 py-1 rounded-lg border border-white/5 backdrop-blur-sm">
                                                <div className="text-[8px] text-blue-200 mb-0.5">سود سازنده</div>
                                                <div className="font-titr text-lg text-green-neon">{res.profitMargin.toFixed(1)} %</div>
                                            </div>
                                        </div>
                                    </div>

                                    {/* 8. Footer */}
